1 YILIAN Temperature tag

Read the temperature tag of Yilian using a dedicated interface ReadTagTemperature。

C Language code example 12-YILIAN Temperature tag


  /**
  *   读宜链温度标签/read Sensor function tag
  */
    #include < stdio.h>
    #include < stdlib.h>
    #include < string.h>
    
    #if defined (__linux__)
    #include < sys/time.h>
    #include < unistd.h>
    #define MUL 1000
    #define MSLEEP(x) usleep(x*MUL)
    #else
    #include < windows.h>
    #define MSLEEP(x) Sleep(x)
    #endif
    
    #include "ModuleReader.h"
    
    time_t startt,endtt;
    int tagcount=0;
    
    int testseconds=10;//测试时间/test time 10s
    
    void PrintErrorDetail(int hreader)
    {
      int internalcode;
      char *errdes;
      char rdrip[50];
      READER_ERR eh_err;
      eh_err =GetReaderAddress(hreader, rdrip);
      eh_err =GetLastDetailError(hreader, &internalcode, &errdes);
      printf("reader:%s ,inerror:%d info:%s\n", rdrip,internalcode, errdes);
    
    }
    
    int main()
    {
      int hreader;
      char* address="192.168.1.160";
      int antportnum=1;
      READER_ERR err;
      int ants[] = {1};
      int tagcnt;
      int timeout=200;
      TAGINFO tag_;
      int i,idx,j;
      int c;
      int isbreak=0;
    
      int ant=1;//操作天线 默认天线口1 /oprate antenna
      unsigned char bank=3; //读温度数据的区/store temperture data bank
      int wordAddress=0x7F; //起始块地址/start word address 
      int wordCount=1; //块数/block count
      int alltime=1100; //总超时时间 这个总超时等于后面两个超时加上通信等待时间/totaol timeout=timeselwait+timereadwait+communication time
      int timeselwait=0; //select超时（温度标签厂商的超时）/select timeout
      int timereadwait=100; //读标签等待超时 / read tags timeout
      unsigned short metaflag=0;//标志/flag
      unsigned char *accesspasswd=NULL; //访问密码/access password
      unsigned char getdata[300];//获取的标签数据 /get tag's data
      int datalen=0;//标签数据的长度/len of tag data
    
      printf("read YILIAN temper tag example.\n");
      err=InitReader_Notype(&hreader, address, antportnum);
      if (err != MT_OK_ERR)
      {
        printf("init reader err :%d\n", err);
        scanf("%c",&c);
        return -1;
      }
    
      err=ReadTagTemperature(hreader,ant,bank,wordAddress,wordCount,alltime,timeselwait,timereadwait,metaflag,accesspasswd,getdata,&datalen);
    
    
      if(err==MT_OK_ERR)
      { 
        unsigned short TAG_METADATA_NONE                = 0x0000;
        unsigned short TAG_METADATA_READCOUNT            = 0x0001;
        unsigned short TAG_METADATA_RSSI                 = 0x0002;
        unsigned short TAG_METADATA_ANTENNAID            = 0x0004;
        unsigned short TAG_METADATA_FREQUENCY            = 0x0008;
        unsigned short TAG_METADATA_TIMESTAMP            = 0x0010; 
        unsigned short TAG_METADATA_PHASE                = 0x0020;
        unsigned short TAG_METADATA_PROTOCOL             = 0x0040;
        unsigned short TAG_EMBEDED_DATA                  = 0X0080;
        int YLtemperture=45;
        int taglen;
        unsigned char tagpc[2];
        unsigned char tagepc[62];
        unsigned char tagcrc[2];
        unsigned char temperdata[2];//wordCount * 2   温度数据/temperture data
    
        //metaflag非0时 数据 return these data when metaflag not 0
        int pvtReadCount;
        int pvtLqi;
        int pvtFrequency;
        int pvtPhase;
        int pvtAntenna;
        unsigned char BankData[64];
        int pvtTsmp;
        int pvtPro;
        float tempet = 0;
    
        int i = 0;
        unsigned char option = getdata[i++];
        if ((option & 0x10) != 0)
        {
          metaflag = (short)(getdata[i++] << 8);
          metaflag |= getdata[i++];
    
          if (0 != (metaflag & TAG_METADATA_READCOUNT))
            pvtReadCount = getdata[i++];
          if (0 != (metaflag & TAG_METADATA_RSSI))
            pvtLqi =  getdata[i++];
          if (0 != (metaflag & TAG_METADATA_ANTENNAID))
          {
            pvtAntenna = getdata[i++] & 0xf;
            if (pvtAntenna == 0)
              pvtAntenna = 16;
          }
          if (0 != (metaflag & TAG_METADATA_FREQUENCY))
          {
            pvtFrequency = (getdata[i++]&0xff) << 16;
            pvtFrequency |= (getdata[i++]&0xff) << 8;
            pvtFrequency |= (getdata[i++]&0xff);
          }
          if (0 != (metaflag & TAG_METADATA_TIMESTAMP))
          {
            pvtTsmp = (getdata[i++]&0xff) << 24;
            pvtTsmp |= (getdata[i++]&0xff) << 16;
            pvtTsmp |=(getdata[i++]&0xff) << 8;
            pvtTsmp |= (getdata[i++]&0xff);
    
          }
          if (0 != (metaflag & TAG_METADATA_PHASE))
          {
            // Ignore this for now
            pvtPhase = getdata[i + 1];
            i += 2;
          }
          if (0 != (metaflag & TAG_METADATA_PROTOCOL))
            pvtPro = getdata[i++];
    
    
          if (0 != (metaflag & TAG_EMBEDED_DATA))
          {
            int bdalen = ((getdata[i] << 8) | getdata[i + 1]) / 8;
            i += 2;
            if (bdalen != 0)
            { 
              memcpy(BankData,getdata+i,bdalen);
            }
            i += bdalen;
          }
        }
    
        idx = i;
        for (i=idx; i < 2+idx; i++)
        {
          temperdata[i - idx] = getdata[i];
        }
    
        taglen = getdata[i++];
        tagpc[0] = getdata[i++];
        tagpc[1] = getdata[i++];
        j=0;
        for (j = i; j < taglen - 4+i; j++)
          tagepc[j-i] = getdata[j];
    
        tagcrc[0] = getdata[i++];
        tagcrc[1] = getdata[i++];
    
    
        if ((temperdata[0] & 0x80) == 0)
        {
          tempet=(temperdata[0]&0xff)-YLtemperture + (temperdata[1]&0xff) * 1.0/ 255;//(temperdata[1]&0xff) * 1/ 255  小数点后部分/decimal part
        }
        else
        {
          temperdata[0] = (byte)(~temperdata[0]);
          temperdata[1] = (byte)(~temperdata[1] + 1);
          tempet = 0-((temperdata[0]&0xff)-YLtemperture + (temperdata[1]&0xff) * 1.0 / 255);
        }
    
      }
      else
      {
        printf("ReadTagTemperature failed err:%d\n", err);
        PrintErrorDetail(hreader);
      }
    
      CloseReader(hreader);
      printf("close the reader.input any key to exit!\n");
      scanf("%c",&c);
      return 0;
    }

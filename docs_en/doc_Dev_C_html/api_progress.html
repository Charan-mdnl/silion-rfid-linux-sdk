Reader lifecycle
 
Firstly, you should callInitReader_Notype function to initialize the reader, and then you can call ParamSet or ParamGet function to configure the reader;Next, you can call tag operations and peripheral functional functions.Call CloseReader function means that you no longer use the reader.
    
    

The main functions and steps are described in the table below：

Execution Steps Functional interface Functional Description
1 InitReader_Notype Connect/Initialize Reader
    
Execution Steps Functional interface Functional Description
2 ParamGet Get Parameters
    
Execution Steps Functional interface Functional Description
2 ParamSet Set Parameters
    
Execution Steps Functional interface Functional Description
3-1 TagInventory_Raw Synchronized inventory
3-2 GetNextTag
    
Execution Steps Functional interface Functional Description
3-1 AsyncStartReading Fast mode invenotry
3-2 AsyncGetTagCount
3-3 AsyncGetNextTag
3-4 AsyncStopReading
    
Execution Steps Functional interface Functional Description
3-1 TagReadHandler,SetTagReadHandler Asynchronous inventory
3-1 ReadingErrorHandler,SetReadingErrorHandler
3-1 GpiStateHandler&GpiTriggerBoundaryHandler,SetGpiTriggerHandler
3-2 StartReading
3-3 StopReading
    
Execution Steps Functional interface Functional Description
3-1 GetTagData read tag bank
3-1 GetTagDataEx read tag bank(impinj QT Functions）
    
Execution Steps Functional interface Functional Description
3-1 WriteTagData write tag bank
3-1 WriteTagEpcEx initialize EPC ID
    
Execution Steps Functional interface Functional Description
3-1 WriteTagData（write access password） lock tag
3-2 LockTag
    
Execution Steps Functional interface Functional Description

3-2 KillTag
    
Execution Steps Functional interface Functional Description

3-1 GetGPIEx

Execution Steps Functional interface Functional Description

C Language Code Example 1- Synchronized Inventory


			/**
			*   普通模式盘点标签10次/genreral mode inventory 10 times
			*/
			#include < stdio.h>
			#include "ModuleReader.h"
			
			void main()
			{
				char c;
				int hreader;
				READER_ERR err;
				int ants[] = {1};
				int tagcnt;
				int timeout=200;
				TAGINFO tag_;
				int i,j;
				int f;
				printf("inventory tags example.\n");
			
				//连接地址为com4,如果指定波特率则可以例如:com4:921600,模块天线端口为1，
				//注意并非已连接天线的数量，指模块的天线端口总数量
				//connect to "com4",if you want to speciate the baudrate,you could input the address as "com4:921600".1 means 1 antenna port.
				//note:the antenna port numbers is all of reader， not the connected antennas numbers.
				err=InitReader_Notype(&hreader, "com4", 1);
				if (err != MT_OK_ERR)
				{
					printf("init the reader err :%d,input any key to exit!\n", err);
					scanf("%c",&c);
					return ;
				}
				printf("init the reader successfully\n");
			
				for(i=0;i< 10;i++)
				{
					err = TagInventory_Raw(hreader, ants, sizeof(ants)/sizeof(ants[0]), timeout, &tagcnt);
					if (err == MT_OK_ERR)
					{
						printf("invenotry ---------------- tagcnt:%d\n", tagcnt);
						
						 for (j = 0; j < tagcnt; ++j)
						 {
							err = GetNextTag(hreader, &tag_);
							//获取缓冲区标签一旦出错，不可以继续获取，重新盘点标签。
							//if failed when called getnexttag,you should inventory again.
							if(err!=MT_OK_ERR)
								break;
			
							 printf("epcid:");
							 for(f = 0; f < tag_.Epclen; ++f)
							 {
								 printf("%02X", tag_.EpcId[f]);
							 }
							printf(" by ant %d, fre:%d rssi:%d\n", tag_.AntennaID, tag_.Frequency,tag_.RSSI-256);
						 }
					}
			
					if(err!=MT_OK_ERR)
					{ 
						printf("inventory tags err:%d\n", err);
					}
				}
			
				printf("close the reader.input any key to exit!\n");
				CloseReader(hreader);
				scanf("%c",&c);
			}

C Language Code Example 2-Read/write tag bank


			/**
			*   读写标签存储区/read and write tag's banks
			*/
			#include < stdio.h>
			#include "ModuleReader.h"
			
			#define BLOCK 2
			void main()
			{
				char c;
				int hreader;
				READER_ERR err;
				int ant= 1;
				int bank=3;
				int startadr=0;
				int timeout=1000;
				int i;
				unsigned char rdata[BLOCK*2];
				unsigned char wdata[BLOCK*2];
				char hexctr[BLOCK*2*2];
				unsigned char accesspwd[4];
			
				printf("read and write tag bank example.\n");
			
				//连接地址为192.168.1.160,默认8080.如果指定端口则可以例如:192.168.1.160:8090,模块天线端口为8，
				//注意并非已连接天线的数量，指模块的天线端口总数量
				//connect to "com4",if you want to speciate the baudrate,you could input the address as "com4:921600".1 means 1 antenna port.
				//note:the antenna port numbers is all of reader， not the connected antennas numbers.
				err=InitReader_Notype(&hreader, "192.168.1.160", 8);
				if (err != 0)
				{
					printf("init the reader err :%d input any key to exit!\n", err);
					scanf("%c",&c);
					return ;
				}
				printf("init the reader successfully\n");
			  
				//标签出厂时，访问密码都是全0,如果标签已经被锁定则根据实际情况设
				//置访问密码
				//the access password is  0 when the tags are manufactured. you should set right password if tag is had locked.
				memset(accesspwd, 0, 4);
				//读标签存储区，超时时间设置为1000ms，也即操作函数最长会阻塞
				//1000ms左右，如果在超时时间以内完成操作或出现错误则会提前返回
				//read to a tag's bank,it will block until be finish.Maximum blocking time is 1000ms.
				err = GetTagData(hreader, ant, bank, startadr, BLOCK, rdata, accesspwd, timeout);
				if (err != MT_OK_ERR)
				{
					printf("GetTagData failed,check whether the tag has user bank. err:%d input any key to exit!\n",err);
					scanf("%c",&c);
				}
			
				Hex2Str(rdata,BLOCK*2,hexctr);
				printf("read the data:%s\n",hexctr);
			
				for(i=0;i< BLOCK*2;i++)
					wdata[i]=rdata[BLOCK*2-1-i];
			
				err = WriteTagData(hreader, ant, bank, startadr, wdata,sizeof(wdata)/sizeof(wdata[0]), accesspwd, timeout);
				if (err != MT_OK_ERR)
				{
					printf("WriteTagData failed, check whether the tag has user bank. err:%d input any key to exit!\n",err);
					scanf("%c",&c);
				}
			
				err = GetTagData(hreader, ant, bank, startadr, BLOCK, rdata, accesspwd, timeout);
				if (err != MT_OK_ERR)
				{
					printf("GetTagData failed, check whether the tag has user bank. err:%d input any key to exit!\n",err);
					scanf("%c",&c);
				}
			
				Hex2Str(rdata,BLOCK*2,hexctr);
				printf("read the data:%s again\n",hexctr);
			
				printf("close the reader.input any key to exit!\n");
				CloseReader(hreader);
				scanf("%c",&c);
			
			}

C Language Code Example 3-Set parameters the get


			/**
			*   读写器参数获取，设置/get parameters form the reader and set parameters to the reader
			*/
			#include < stdio.h>
			#include "ModuleReader.h"
			
			void main()
			{
				char c;
				int hreader;
				char* address="192.168.1.160";
				int antportnum=8;
				READER_ERR err;
				unsigned short Maxp;
				unsigned short Minp;
				unsigned short Wantp=2000;//20dbm
				AntPowerConf antpwrconf,antpwrconf2;
				int m_qval;
				int session;
				int target;
				int rfmode;
				int m_qval2;
				int session2;
				int target2;
				int rfmode2;
				Region_Conf region;
				int i,j;
				ConnAnts_ST conants;
				HoptableData_ST hoptable,hoptable2;
				printf("set and get parameter example.\n");
			
				//连接地址为192.168.1.160,默认8080.如果指定端口则可以例如:192.168.1.160:8090,模块天线端口为8，
				//注意并非已连接天线的数量，指模块的天线端口总数量
				//connect to "com4",if you want to speciate the baudrate,you could input the address as "com4:921600".1 means 1 antenna port.
				//note:the antenna port numbers is all of reader， not the connected antennas numbers.
				err=InitReader_Notype(&hreader, address, antportnum);
				if (err != 0)
				{
					printf("init the reader err :%d input any key to exit!\n", err);
					scanf("%c",&c);
					return ;
				}
				printf("init the reader successfully\n");
			
			
				if ((err=ParamGet(hreader, MTR_PARAM_RF_MAXPOWER, &Maxp)) != MT_OK_ERR)
				{
					printf("ParamGet err :%d input any key to exit!\n", err);
					scanf("%c",&c);
			
				}
				if ((err=ParamGet(hreader, MTR_PARAM_RF_MINPOWER, &Minp)) != MT_OK_ERR)
				{
					printf("ParamGet err :%d input any key to exit!\n", err);
					scanf("%c",&c);
			
				}
			
				if(Wantp>Maxp||Wantp< Minp)
				{
					printf("power value err. input any key to exit!\n");
					scanf("%c",&c);
					CloseReader(hreader);
					return;
				}
			
				//设置功率，功率越大天线覆盖面越大。
				//设置功率结构体
				//set power more large more the radiation range.
				//set to the power struct
				antpwrconf.antcnt = antportnum;
				for(i=0;i< antpwrconf.antcnt;i++)
				{
					antpwrconf.Powers[i].antid = i+1; //天线号从1开始/ antenna id start since 1
					antpwrconf.Powers[i].readPower = Wantp; //读功率 /read power
					antpwrconf.Powers[i].writePower = Wantp; //写功率 /write power
				}
				if ((err=ParamSet(hreader, MTR_PARAM_RF_ANTPOWER, &antpwrconf)) != MT_OK_ERR)
				{
					printf("ParamSet MTR_PARAM_RF_ANTPOWER err:%d  input any key to exit!\n",err);
					scanf("%c",&c);
			
				}
			
				if ((err=ParamGet(hreader, MTR_PARAM_RF_ANTPOWER, &antpwrconf2)) != MT_OK_ERR)
				{
					printf("ParamGet MTR_PARAM_RF_ANTPOWER err:%d  input any key to exit!\n",err);
					scanf("%c",&c);
			
				}
			
				m_qval=5;
				if ((err=ParamSet(hreader, MTR_PARAM_POTL_GEN2_Q, &m_qval)) != MT_OK_ERR)
				{
					printf("ParamSet MTR_PARAM_POTL_GEN2_Q err:%d  input any key to exit!\n",err);
					scanf("%c",&c);
			
				}
				if ((err=ParamGet(hreader, MTR_PARAM_POTL_GEN2_Q, &m_qval2)) != MT_OK_ERR)
				{
					printf("ParamGet MTR_PARAM_POTL_GEN2_Q err:%d  input any key to exit!\n",err);
					scanf("%c",&c);
			
				}
				printf("ParamGet MTR_PARAM_POTL_GEN2_Q :%d\n",m_qval2);
			
				session=1;
				if ((err=ParamSet(hreader, MTR_PARAM_POTL_GEN2_SESSION, &session)) != MT_OK_ERR)
				{
					printf("ParamSet MTR_PARAM_POTL_GEN2_SESSION err:%d  input any key to exit!\n",err);
					scanf("%c",&c);
			
				}
				if ((err=ParamGet(hreader, MTR_PARAM_POTL_GEN2_SESSION, &session2)) != MT_OK_ERR)
				{
					printf("ParamGet MTR_PARAM_POTL_GEN2_SESSION err:%d  input any key to exit!\n",err);
					scanf("%c",&c);
			
				}
				printf("ParamGet MTR_PARAM_POTL_GEN2_SESSION :%d\n",session2);
			
				target=2;
				if ((err=ParamSet(hreader, MTR_PARAM_POTL_GEN2_TARGET, &target)) != MT_OK_ERR)
				{
					printf("ParamSet MTR_PARAM_POTL_GEN2_TARGET err:%d  input any key to exit!\n",err);
					scanf("%c",&c);
			
				}
				if ((err=ParamGet(hreader, MTR_PARAM_POTL_GEN2_TARGET, &target2)) != MT_OK_ERR)
				{
					printf("ParamGet MTR_PARAM_POTL_GEN2_TARGET err:%d  input any key to exit!\n",err);
					scanf("%c",&c);
			
				}
				printf("ParamGet MTR_PARAM_POTL_GEN2_TARGET :%d\n",target2);
			
				rfmode=105;
				if ((err=ParamSet(hreader, MTR_PARAM_POTL_GEN2_TAGENCODING, &rfmode)) != MT_OK_ERR)
				{
					printf("ParamSet MTR_PARAM_POTL_GEN2_TAGENCODING err:%d  input any key to exit!\n",err);
					scanf("%c",&c);
			
				}

				{

					scanf("%c",&c);
			
				}
				printf("ParamGet MTR_PARAM_POTL_GEN2_TAGENCODING :%d\n",rfmode2);
			
			
				if ((err=ParamGet(hreader, MTR_PARAM_READER_CONN_ANTS, &conants)) != MT_OK_ERR)
				{
					printf("ParamGet MTR_PARAM_READER_CONN_ANTS err:%d  input any key to exit!\n",err);
					scanf("%c",&c);
					CloseReader(hreader);
					return;
				}
				printf("ParamGet MTR_PARAM_READER_CONN_ANTS :");
				for(i=0;i< conants.antcnt;i++)
					printf(" %d",conants.connectedants[i]);
				printf("\n");
			
			
				region=RG_PRC;
				if ((err=ParamSet(hreader, MTR_PARAM_FREQUENCY_REGION, & region)) != MT_OK_ERR)
				{
					printf("ParamSet MTR_PARAM_FREQUENCY_REGION err:%d  input any key to exit!\n",err);

				hoptable2.lenhtb=hoptable.lenhtb/2;

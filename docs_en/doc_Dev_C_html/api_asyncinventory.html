Asynchronous inventory

The API implements asynchronous inventory and uses the StartReading interface function to start asynchronous inventory.Asynchronous inventory implements two types of inventory interface calls internally, switching between normal inventory and fast mode inventory methods through the IsFastRead property in the parameter BackReadOption. The normal inventory mode is a type of asynchronous inventory mode that involves creating a new thread internally to call the synchronous inventory TagInventory_Raw function.This asynchronous inventory is called the normal asynchronous inventory mode. Another type is true asynchronous inventory, which calls the AsyncStartReading function in fast mode inventory internally, and the reader will remain in a continuous inventory state.This method has the best inventory performance for reader. This method has the best inventory performance for readers. For applications with high inventory performance requirements, this asynchronous inventory mode should be used, which is called the fast asynchronous inventory mode. Stop asynchronous inventory using the StopReading interface function.

Note: Not all device types support fast asynchronous inventory mode. R2000, E-series, SFM series and other chip based readers support this inventory mode.

The asynchronous inventory returns tag data and exceptions through callback. Before calling StartReading, call the interface function SetTagReadHandler to set the tag callback; If tags are inventoried during asynchronous inventory, the TagReadHandler will be called back to pass the inventoried tag data to the user program; Call the interface function SetReadingErrorHandler to set an exception callback before calling StartReading;If the reader encounters an abnormal situation, it will call ReadingErrorHandler to pass the detailed information of the exception to the user program;Call the interface function SetGpiTriggerHandler before calling StartReading to set the GPI trigger callback; If asynchronous inventory is configured to start or stop triggered by GPI, then GpiStateHandler and GpiTriggerBoundaryHandler will be called back when GPI conditions are met. Enable GPI triggering through the IsGPITrigger property of BackReadOption.

    int main()  
    {
      int hreader;
      char* address="192.168.1.160";
      int antportnum=1;
      READER_ERR err;
      int ants[] = {1};
      int *pants=&ants[0];
      int antlen=sizeof(ants)/sizeof(ants[0]);
      int i;
      unsigned short Maxp;
      CustomParam_ST cpara;
      AntPowerConf antpwrconf;
      int c;
      ConnAnts_ST conants;
    
      unsigned char  paraBuf[50+22]="Reader/Ex10fastmode";

      OnReadingErrorBlock reblk;
      OnTagReadBlock trblk;
      BackReadOption BROption;
    
      printf("fastmode AsynRead example .\n");
      err=InitReader_Notype(&hreader, address, antportnum);
      if (err != MT_OK_ERR)
      {
        printf("init reader err :%d\n", err);
        scanf("%c",&c);
        return -1;
      }
    
      if ((err=ParamGet(hreader, MTR_PARAM_RF_MAXPOWER, &Maxp)) != MT_OK_ERR)
      {
        printf("ParamGet err :%d input any key to exit!\n", err);
        scanf("%c",&c);
        return ;
      }
    
      antpwrconf.antcnt = antportnum;
      for(i=0;i< antportnum;i++)
      { antpwrconf.Powers[i].antid = i+1;
      antpwrconf.Powers[i].readPower = Maxp;
      antpwrconf.Powers[i].writePower = Maxp;}

      if (ParamSet(hreader, MTR_PARAM_RF_ANTPOWER, &antpwrconf) != MT_OK_ERR)
      {
        printf("ParamSet MTR_PARAM_RF_ANTPOWER err\n");
        scanf("%c",&c);
        return -1;
      }
     
      //获取已连接天线/get connected antennnas
      if (ParamGet(hreader, MTR_PARAM_READER_CONN_ANTS, &conants) == MT_OK_ERR)
      {
        if(conants.antcnt>0)
        { 
          printf("get connected antennas \n");
          pants=&conants.connectedants[0];
          antlen=conants.antcnt;
        }
      }
    
      reblk.handler = OnReadingError;
      reblk.cookie = NULL;
      SetReadingErrorHandler(hreader, reblk);
    
      trblk.handler = OnTagRead;
      trblk.cookie = NULL;
      SetTagReadHandler(hreader, trblk);
    
      memset(&BROption, 0, sizeof(BackReadOption));
      BROption.IsFastRead = 1; //0 普通模式 1 快速模式 /0 general mode 1 fast 
    
      paraBuf[50]=0;// 1 enable ex10fastmode，0 genral fast mode
      paraBuf[51]=20;
      for(i=52;i< paraBuf[51];i++)
        paraBuf[i]=0;
      cpara.pCusParam =paraBuf;
      cpara.CParamlen=22;
      err = ParamSet(hreader,MTR_PARAM_CUSTOM, &cpara);
    
      BROption.TMFlags.IsAntennaID=1;
      BROption.TMFlags.IsRSSI=1;
    
      err = StartReading(hreader, pants, antlen, &BROption);
      if (err == MT_OK_ERR)
      {	printf("StartReading successfully\n");
    
      startt=time((time_t*)NULL);
      endtt=startt;}
      else
      {
        printf("StartReading failed err:%d\n", err);
        scanf("%c",&c);
        return -1;
      }
     
      while (1)
      {
        endtt=time((time_t*)NULL);
        if((endtt-startt)>=testseconds||isbreak)
        {
          StopReading(hreader);
          break;
        }
        MSLEEP(1000);
      }
    
      CloseReader(hreader);
      printf("close the reader.input any key to exit!\n");
      scanf("%c",&c);
      return 0;
    }

      if (err == MT_OK_ERR)
      {	printf("StartReading successfully\n");
      startt=time((time_t*)NULL);

        return -1;
      }
      while (1)

        endtt=time((time_t*)NULL);
        if((endtt-startt)>=testseconds||isbreak)
        {
          StopReading(hreader);
          break;
        }
        MSLEEP(1000);
      }

      CloseReader(hreader);
      printf("close the reader.input any key to exit!\n");
      scanf("%c",&c);
      return 0;
    }
